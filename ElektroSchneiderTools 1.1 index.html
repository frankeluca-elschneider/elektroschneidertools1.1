<script>

    // Navigation

    function showPage(id) {

        document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));

        document.getElementById(id).classList.add('active');

        window.scrollTo(0,0);

    }


    // VDE Checkliste Generierung

    const fragen = ["Kennzeichnung der Kreise", "Schutz gegen Berühren", "Leitungsverlegung", "Rechtsdrehfeld", "PE-Anschluss"];

    const list = document.getElementById('vde-list');

    fragen.forEach(f => {

        list.innerHTML += `<div style="display:flex; justify-content:space-between; align-items:center; margin-bottom:12px;">

            <span style="font-size:0.9rem;">${f}</span>

            <select style="width:auto; padding:5px;"><option>i.O.</option><option>n.i.O.</option></select>

        </div>`;

    });


    // Unterschriften-Feld Logik

    const canvas = document.getElementById('signature-pad');

    const ctx = canvas.getContext('2d');

    let isDrawing = false;


    function getMousePos(e) {

        const rect = canvas.getBoundingClientRect();

        const clientX = e.clientX || e.touches[0].clientX;

        const clientY = e.clientY || e.touches[0].clientY;

        return { x: clientX - rect.left, y: clientY - rect.top };

    }


    const startDrawing = (e) => { isDrawing = true; const pos = getMousePos(e); ctx.moveTo(pos.x, pos.y); };

    const draw = (e) => { if (!isDrawing) return; e.preventDefault(); const pos = getMousePos(e); ctx.lineTo(pos.x, pos.y); ctx.stroke(); };

    const stopDrawing = () => { isDrawing = false; ctx.beginPath(); };


    canvas.addEventListener('mousedown', startDrawing);

    canvas.addEventListener('mousemove', draw);

    window.addEventListener('mouseup', stopDrawing);

    canvas.addEventListener('touchstart', startDrawing);

    canvas.addEventListener('touchmove', draw);

    canvas.addEventListener('touchend', stopDrawing);


    function clearSig() { ctx.clearRect(0, 0, canvas.width, canvas.height); }


    // PDF Export

    function savePDF() {

        const { jsPDF } = window.jspdf;

        const doc = new jsPDF();

        doc.setFontSize(18);

        doc.text("Elektro Schneider Tools - VDE 0100", 20, 20);

        doc.setFontSize(12);

        doc.text("Kunde: " + document.getElementById('vde-kunde').value, 20, 40);

        doc.text("Anlage: " + document.getElementById('vde-id').value, 20, 50);

        doc.addImage(canvas.toDataURL(), 'PNG', 20, 70, 60, 25);

        doc.save("VDE_Protokoll.pdf");

    }


    // WhatsApp

    function sendWhatsApp() {

        const fz = document.getElementById('fz-id').value;

        const fehlt = document.getElementById('fz-fehlt').value;

        const msg = `*Elektro Schneider FZ-Bericht*%0A*Fahrzeug:* ${fz}%0A*Bedarf:* ${fehlt}`;

        window.open(`https://wa.me/?text=${msg}`, '_blank');

    }
